CREATE DEFINER=`ineqbench_user`@`%` PROCEDURE `getUnpaidCareOutput`()
BEGIN
	SELECT
	(SELECT
		REPLACE(FORMAT(SUM(TOTAL),0),',','')
		FROM UNPAID_CARE) AS 'totalPopulation',
    (SELECT
		REPLACE(FORMAT((SUM('1-19_HOURS_CARE')+SUM('20-49_HOURS_CARE')+SUM('50_PLUS_HOURS_CARE')),0),',','')
		FROM UNPAID_CARE) AS 'totalDeprived';
END
